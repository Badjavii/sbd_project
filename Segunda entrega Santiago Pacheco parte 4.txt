CREATE OR REPLACE PROCEDURE sp_generar_calendario (
    p_id_club        IN NUMBER,
    p_numero_grupo   IN NUMBER,
    p_isbn_libro     IN NUMBER,
    p_id_moderador   IN NUMBER,
    p_mes            IN NUMBER,
    p_anno           IN NUMBER
) AS
    v_fecha_inicio       DATE;
    v_fecha_fin          DATE;
    v_fecha_actual       DATE;
    v_reuniones_creadas  NUMBER := 0;
    v_categoria_edad     VARCHAR2(10);

    e_regla_negocio      EXCEPTION;
    v_msg_error          VARCHAR2(300);
BEGIN
    SELECT categoria_edad INTO v_categoria_edad
    FROM SOJG_GRUPO_DE_LECTURA
    WHERE id_club = p_id_club AND numero_de_grupo = p_numero_grupo;

    v_fecha_inicio := TO_DATE('01/' || TO_CHAR(p_mes, 'FM00') || '/' || TO_CHAR(p_anno), 'DD/MM/YYYY');
    v_fecha_fin    := LAST_DAY(v_fecha_inicio);
    v_fecha_actual := v_fecha_inicio;

    WHILE v_fecha_actual <= v_fecha_fin AND v_reuniones_creadas < 3 LOOP
            IF TO_CHAR(v_fecha_actual, 'D') NOT IN ('1', '7') THEN

                v_reuniones_creadas := v_reuniones_creadas + 1;

                INSERT INTO SOJG_CALENDARIO_MES (
                    id_club,
                    numero_de_grupo,
                    isbn_libro,
                    fecha_reunion,
                    realizada,
                    ultima_discusion,
                    id_moderador
                ) VALUES (
                             p_id_club,
                             p_numero_grupo,
                             p_isbn_libro,
                             v_fecha_actual,
                             'NO',
                             CASE WHEN v_reuniones_creadas = 3 THEN 'SI' ELSE 'NO' END,
                             p_id_moderador
                         );

            END IF;
            v_fecha_actual := v_fecha_actual + 1;
        END LOOP;

    IF v_reuniones_creadas < 3 THEN
        v_msg_error := 'Error: El mes solicitado no cuenta con suficientes días hábiles para programar las 3 reuniones.';
        RAISE e_regla_negocio;
    END IF;

    COMMIT;
    DBMS_OUTPUT.PUT_LINE('Calendario generado con éxito. 3 reuniones registradas.');

EXCEPTION
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('Error: El club o el grupo especificado no existe.');
        ROLLBACK;
    WHEN e_regla_negocio THEN
        DBMS_OUTPUT.PUT_LINE(v_msg_error);
        ROLLBACK;
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('Error inesperado: ' || SQLERRM);
        ROLLBACK;
END;
/

CREATE OR REPLACE PROCEDURE sp_asignar_moderador (
    p_id_club        IN NUMBER,
    p_numero_grupo   IN NUMBER,
    p_isbn_libro     IN NUMBER,
    p_id_moderador   IN NUMBER
) AS
    v_categoria_grupo      VARCHAR2(10);
    v_moderador_ocupado    NUMBER := 0;
    v_es_adulto_del_club   NUMBER := 0;
    v_pertenece_grupo      NUMBER := 0;

    e_regla_negocio        EXCEPTION;
    v_msg_error            VARCHAR2(300);
BEGIN
    SELECT categoria_edad INTO v_categoria_grupo
    FROM SOJG_GRUPO_DE_LECTURA
    WHERE id_club = p_id_club AND numero_de_grupo = p_numero_grupo;

    SELECT COUNT(*) INTO v_moderador_ocupado
    FROM SOJG_CALENDARIO_MES
    WHERE id_moderador = p_id_moderador
      AND realizada = 'NO'
      AND (id_club != p_id_club OR numero_de_grupo != p_numero_grupo);

    IF v_moderador_ocupado > 0 THEN
        v_msg_error := 'Error: El miembro ya está asignado como moderador en otro grupo con discusiones activas.';
        RAISE e_regla_negocio;
    END IF;

    IF v_categoria_grupo = 'Nino' THEN
        SELECT COUNT(*) INTO v_es_adulto_del_club
        FROM SOJG_HISTORICO_GRUPO_LECTURA hgl
                 JOIN SOJG_GRUPO_DE_LECTURA gl
                      ON (hgl.id_club = gl.id_club AND hgl.numero_de_grupo = gl.numero_de_grupo)
        WHERE hgl.id_miembro = p_id_moderador
          AND hgl.id_club = p_id_club
          AND gl.categoria_edad = 'Adulto'
          AND hgl.fecha_fin IS NULL;

        IF v_es_adulto_del_club = 0 THEN
            v_msg_error := 'Error de Seguridad: Los grupos de Niños solo pueden ser moderados por un miembro Adulto del mismo club.';
            RAISE e_regla_negocio;
        END IF;
    ELSE
        SELECT COUNT(*) INTO v_pertenece_grupo
        FROM SOJG_HISTORICO_GRUPO_LECTURA
        WHERE id_miembro = p_id_moderador
          AND id_club = p_id_club
          AND numero_de_grupo = p_numero_grupo
          AND fecha_fin IS NULL;

        IF v_pertenece_grupo = 0 THEN
            v_msg_error := 'Error: El moderador seleccionado debe ser un miembro activo de este grupo de lectura.';
            RAISE e_regla_negocio;
        END IF;
    END IF;

    UPDATE SOJG_CALENDARIO_MES
    SET id_moderador = p_id_moderador
    WHERE id_club = p_id_club
      AND numero_de_grupo = p_numero_grupo
      AND isbn_libro = p_isbn_libro;

    IF SQL%ROWCOUNT = 0 THEN
        v_msg_error := 'Error: No se encontraron reuniones programadas para el libro especificado.';
        RAISE e_regla_negocio;
    END IF;

    COMMIT;
    DBMS_OUTPUT.PUT_LINE('Moderador asignado exitosamente.');

EXCEPTION
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('Error: El club o grupo especificado no existe.');
        ROLLBACK;
    WHEN e_regla_negocio THEN
        DBMS_OUTPUT.PUT_LINE(v_msg_error);
        ROLLBACK;
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('Error inesperado: ' || SQLERRM);
        ROLLBACK;
END;
/

CREATE OR REPLACE PROCEDURE sp_registrar_inasistencia (
    p_id_club        IN NUMBER,
    p_numero_grupo   IN NUMBER,
    p_isbn_libro     IN NUMBER,
    p_fecha_reunion  IN DATE,
    p_id_miembro     IN NUMBER
) AS
    v_reunion_realizada    VARCHAR2(2);
    v_pertenece_grupo      NUMBER := 0;
    v_ya_registrado        NUMBER := 0;

    e_regla_negocio        EXCEPTION;
    v_msg_error            VARCHAR2(300);
BEGIN
    BEGIN
        SELECT realizada INTO v_reunion_realizada
        FROM SOJG_CALENDARIO_MES
        WHERE id_club = p_id_club
          AND numero_de_grupo = p_numero_grupo
          AND isbn_libro = p_isbn_libro
          AND fecha_reunion = p_fecha_reunion;
    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            v_msg_error := 'Error: La reunión especificada no existe en el calendario.';
            RAISE e_regla_negocio;
    END;

    IF v_reunion_realizada = 'NO' THEN
        v_msg_error := 'Error: No se puede registrar una inasistencia en una reunión que no ha sido realizada.';
        RAISE e_regla_negocio;
    END IF;

    SELECT COUNT(*) INTO v_pertenece_grupo
    FROM SOJG_HISTORICO_GRUPO_LECTURA
    WHERE id_miembro = p_id_miembro
      AND id_club = p_id_club
      AND numero_de_grupo = p_numero_grupo
      AND fecha_inicio <= p_fecha_reunion
      AND (fecha_fin IS NULL OR fecha_fin >= p_fecha_reunion);

    IF v_pertenece_grupo = 0 THEN
        v_msg_error := 'Error: El miembro no estaba activo en este grupo de lectura durante la fecha de la reunión.';
        RAISE e_regla_negocio;
    END IF;

    SELECT COUNT(*) INTO v_ya_registrado
    FROM SOJG_INASISTENCIA
    WHERE id_club = p_id_club
      AND numero_de_grupo = p_numero_grupo
      AND isbn_libro = p_isbn_libro
      AND fecha_reunion = p_fecha_reunion
      AND id_miembro = p_id_miembro;

    IF v_ya_registrado > 0 THEN
        v_msg_error := 'Aviso: La inasistencia ya se encontraba registrada.';
        RAISE e_regla_negocio;
    END IF;

    INSERT INTO SOJG_INASISTENCIA (
        id_club,
        numero_de_grupo,
        isbn_libro,
        fecha_reunion,
        id_miembro
    ) VALUES (
                 p_id_club,
                 p_numero_grupo,
                 p_isbn_libro,
                 p_fecha_reunion,
                 p_id_miembro
             );

    COMMIT;
    DBMS_OUTPUT.PUT_LINE('Inasistencia registrada con éxito.');

EXCEPTION
    WHEN e_regla_negocio THEN
        DBMS_OUTPUT.PUT_LINE(v_msg_error);
        ROLLBACK;
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('Error inesperado: ' || SQLERRM);
        ROLLBACK;
END;
/

CREATE OR REPLACE PROCEDURE sp_cerrar_discusion (
    p_id_club        IN NUMBER,
    p_numero_grupo   IN NUMBER,
    p_isbn_libro     IN NUMBER,
    p_conclusiones   IN VARCHAR2,
    p_valoracion     IN NUMBER
) AS
    v_existe_reunion   NUMBER := 0;

    e_regla_negocio    EXCEPTION;
    PRAGMA EXCEPTION_INIT(e_regla_negocio, -2290); -- Captura el error de CHECK constraint de Oracle

    v_msg_error        VARCHAR2(300);
BEGIN
    -- 1. VALIDACIÓN: Verificar si existe la reunión final para este libro en el grupo
    SELECT COUNT(*) INTO v_existe_reunion
    FROM SOJG_CALENDARIO_MES
    WHERE id_club = p_id_club
      AND numero_de_grupo = p_numero_grupo
      AND isbn_libro = p_isbn_libro
      AND ultima_discusion = 'SI';

    IF v_existe_reunion = 0 THEN
        v_msg_error := 'Error: No se encontró la reunión final de discusión para el libro y grupo especificados.';
        RAISE_APPLICATION_ERROR(-20001, v_msg_error);
    END IF;

    -- 2. EJECUCIÓN: Actualizar los campos de cierre en la última reunión del libro
    UPDATE SOJG_CALENDARIO_MES
    SET conclusiones = p_conclusiones,
        valoracion_final = p_valoracion,
        realizada = 'SI' -- Al cerrar la discusión, la reunión se marca como efectuada
    WHERE id_club = p_id_club
      AND numero_de_grupo = p_numero_grupo
      AND isbn_libro = p_isbn_libro
      AND ultima_discusion = 'SI';

    COMMIT;
    DBMS_OUTPUT.PUT_LINE('Discusión del libro ' || p_isbn_libro || ' cerrada exitosamente con una valoración de ' || p_valoracion || '.');

EXCEPTION
    WHEN e_regla_negocio THEN
        -- Si salta por el CHECK de valoracion_final (between 1 and 5)
        DBMS_OUTPUT.PUT_LINE('Error: La valoración debe ser un número decimal o entero entre 1.0 y 5.0.');
        ROLLBACK;
    WHEN OTHERS THEN
        IF SQLCODE = -20001 THEN
            DBMS_OUTPUT.PUT_LINE(SQLERRM);
        ELSE
            DBMS_OUTPUT.PUT_LINE('Error inesperado al cerrar discusión: ' || SQLERRM);
        END IF;
        ROLLBACK;
END;
/

CREATE OR REPLACE PROCEDURE sp_split_grupo (
    p_id_club           IN NUMBER,
    p_numero_grupo_orig IN NUMBER
) AS
    v_nuevo_grupo        NUMBER;
    v_total_miembros     NUMBER := 0;
    v_mitad              NUMBER := 0;
    v_contador           NUMBER := 0;

    -- Variables para copiar la configuración del grupo original
    v_categoria_edad     VARCHAR2(10);
    v_dia_reunion        VARCHAR2(15);
    v_hora_inicio        NUMBER(4,2);

    -- Cursor para obtener los miembros activos ordenados por antigüedad (antiguos primero)
    CURSOR c_miembros IS
        SELECT id_miembro, fecha_inicio_membresia, fecha_inicio
        FROM SOJG_HISTORICO_GRUPO_LECTURA
        WHERE id_club = p_id_club
          AND numero_de_grupo = p_numero_grupo_orig
          AND fecha_fin IS NULL
        ORDER BY fecha_inicio ASC;

    e_regla_negocio      EXCEPTION;
    v_msg_error          VARCHAR2(300);
BEGIN
    -- 1. Obtener los datos de configuración del grupo original
    SELECT categoria_edad, dia_reunion, hora_inicio
    INTO v_categoria_edad, v_dia_reunion, v_hora_inicio
    FROM SOJG_GRUPO_DE_LECTURA
    WHERE id_club = p_id_club AND numero_de_grupo = p_numero_grupo_orig;

    -- 2. Contar cuántos miembros activos tiene el grupo actualmente
    SELECT COUNT(*) INTO v_total_miembros
    FROM SOJG_HISTORICO_GRUPO_LECTURA
    WHERE id_club = p_id_club
      AND numero_de_grupo = p_numero_grupo_orig
      AND fecha_fin IS NULL;

    -- Validación: No tiene sentido dividir un grupo con muy pocos miembros
    IF v_total_miembros < 2 THEN
        v_msg_error := 'Error: El grupo no tiene suficientes miembros activos para realizar una división.';
        RAISE e_regla_negocio;
    END IF;

    -- Calcular cuántos miembros se van a mudar (la mitad más nueva)
    v_mitad := TRUNC(v_total_miembros / 2);

    -- 3. Calcular el identificador del nuevo grupo secuencial para ese club
    SELECT NVL(MAX(numero_de_grupo), 0) + 1 INTO v_nuevo_grupo
    FROM SOJG_GRUPO_DE_LECTURA
    WHERE id_club = p_id_club;

    -- 4. Crear el nuevo grupo espejo en el catálogo
    INSERT INTO SOJG_GRUPO_DE_LECTURA (
        id_club,
        numero_de_grupo,
        fecha_de_creacion,
        categoria_edad,
        dia_reunion,
        hora_inicio
    ) VALUES (
                 p_id_club,
                 v_nuevo_grupo,
                 SYSDATE,
                 v_categoria_edad,
                 v_dia_reunion,
                 v_hora_inicio
             );

    -- 5. Recorrer el cursor para mudar a los miembros más recientes
    FOR r_miembro IN c_miembros LOOP
            v_contador := v_contador + 1;

            -- Los primeros registros corresponden a los más antiguos (se quedan en el original)
            -- Cuando el contador supera la mitad, procesamos la mudanza de los más nuevos
            IF v_contador > (v_total_miembros - v_mitad) THEN

                -- A) Cerrar la asignación en el grupo viejo
                UPDATE SOJG_HISTORICO_GRUPO_LECTURA
                SET fecha_fin = SYSDATE
                WHERE id_club = p_id_club
                  AND numero_de_grupo = p_numero_grupo_orig
                  AND id_miembro = r_miembro.id_miembro
                  AND fecha_inicio = r_miembro.fecha_inicio;

                -- B) Abrir la asignación en el grupo nuevo
                INSERT INTO SOJG_HISTORICO_GRUPO_LECTURA (
                    id_club,
                    numero_de_grupo,
                    fecha_inicio_membresia,
                    id_miembro,
                    fecha_inicio,
                    fecha_fin
                ) VALUES (
                             p_id_club,
                             v_nuevo_grupo,
                             r_miembro.fecha_inicio_membresia,
                             r_miembro.id_miembro,
                             SYSDATE,
                             NULL
                         );

            END IF;
        END LOOP;

    COMMIT;
    DBMS_OUTPUT.PUT_LINE('Split ejecutado con éxito. Creado el grupo ' || v_nuevo_grupo || '. Se mudaron ' || v_mitad || ' miembros.');

EXCEPTION
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('Error: El grupo de origen no existe.');
        ROLLBACK;
    WHEN e_regla_negocio THEN
        DBMS_OUTPUT.PUT_LINE(v_msg_error);
        ROLLBACK;
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('Error inesperado en el proceso de Split: ' || SQLERRM);
        ROLLBACK;
END;
/

CREATE OR REPLACE TRIGGER trg_bloquear_split_en_discusion
    BEFORE INSERT OR UPDATE ON SOJG_HISTORICO_GRUPO_LECTURA
    FOR EACH ROW
DECLARE
    v_reuniones_pendientes NUMBER := 0;
BEGIN
    -- Verificar si el grupo de lectura tiene discusiones activas (libros no cerrados)
    SELECT COUNT(*) INTO v_reuniones_pendientes
    FROM SOJG_CALENDARIO_MES
    WHERE id_club = :NEW.id_club
      AND numero_de_grupo = :NEW.numero_de_grupo
      AND realizada = 'NO';

    -- Si existen reuniones pendientes, se bloquea cualquier mutación en el grupo
    IF v_reuniones_pendientes > 0 THEN
        RAISE_APPLICATION_ERROR(-20002,
                                'Operación Bloqueada: No se pueden alterar los miembros ni dividir el grupo '
                                    || :NEW.numero_de_grupo || ' porque tiene una discusión de libro activa.');
    END IF;
END;
/

CREATE OR REPLACE TRIGGER trg_validar_moderador_ninos
    BEFORE INSERT OR UPDATE OF id_moderador ON SOJG_CALENDARIO_MES
    FOR EACH ROW
DECLARE
    v_categoria_grupo    VARCHAR2(10);
    v_es_adulto_del_club NUMBER := 0;
BEGIN
    -- 1. Consultar la categoría del grupo de la reunión que se está afectando
    SELECT categoria_edad INTO v_categoria_grupo
    FROM SOJG_GRUPO_DE_LECTURA
    WHERE id_club = :NEW.id_club
      AND numero_de_grupo = :NEW.numero_de_grupo;

    -- 2. Si el grupo es de niños, validar las credenciales del moderador entrante
    IF v_categoria_grupo = 'Nino' THEN
        SELECT COUNT(*) INTO v_es_adulto_del_club
        FROM SOJG_HISTORICO_GRUPO_LECTURA hgl
                 JOIN SOJG_GRUPO_DE_LECTURA gl
                      ON (hgl.id_club = gl.id_club AND hgl.numero_de_grupo = gl.numero_de_grupo)
        WHERE hgl.id_miembro = :NEW.id_moderador
          AND hgl.id_club = :NEW.id_club
          AND gl.categoria_edad = 'Adulto'
          AND hgl.fecha_fin IS NULL;

        IF v_es_adulto_del_club = 0 THEN
            RAISE_APPLICATION_ERROR(-20003,
                                    'Violación de Seguridad: El moderador asignado (' || :NEW.id_moderador ||
                                    ') no es un miembro activo de un grupo Adulto en este club.');
        END IF;
    END IF;
END;
/

CREATE OR REPLACE VIEW vw_reporte_financiero AS
SELECT
    c.id_club,
    c.nombre AS nombre_club,
    ci.nombre AS ciudad,
    EXTRACT(YEAR FROM p.fecha) AS anno,
    -- Total de ingresos por membresías
    NVL(SUM(p.monto), 0) AS total_ingresos_membresia,
    -- Subconsulta correlacionada para la recaudación de obras de teatro del mismo club y año
    NVL((
            SELECT SUM(o.costo_entrada * pres.cantidad_asistentes)
            FROM SOJG_OBRA o
                     JOIN SOJG_PRESENTACION_DE_LA_OBRA pres ON o.id_obra = pres.id_obra
            WHERE o.id_club = c.id_club
              AND EXTRACT(YEAR FROM pres.fecha) = EXTRACT(YEAR FROM p.fecha)
        ), 0) AS total_recaudacion_obras,
    -- Subconsulta para contar el total de votos emitidos para actores en ese año y club
    NVL((
            SELECT SUM(ma.votos)
            FROM SOJG_OBRA o
                     JOIN SOJG_MEJOR_ACTOR ma ON o.id_obra = ma.id_obra
            WHERE o.id_club = c.id_club
              AND EXTRACT(YEAR FROM ma.fecha_presentacion) = EXTRACT(YEAR FROM p.fecha)
        ), 0) AS total_votos_actores
FROM SOJG_CLUB c
         JOIN SOJG_CIUDAD ci ON c.id_ciudad = ci.id
         LEFT JOIN SOJG_PAGO_MEMBRESIA p ON c.id_club = p.id_club
GROUP BY c.id_club, c.nombre, ci.nombre, EXTRACT(YEAR FROM p.fecha);
/

SELECT
    id_club AS "ID CLUB",
    nombre_club AS "CLUB DE LECTURA",
    ciudad AS "CIUDAD",
    anno AS "AÑO FISCAL",
    TO_CHAR(total_ingresos_membresia, '$999,999,990.00') AS "INGRESOS MEMBRESÍAS",
    TO_CHAR(total_recaudacion_obras, '$999,999,990.00') AS "RECAUDACIÓN TAQUILLA",
    TO_CHAR(total_ingresos_membresia + total_recaudacion_obras, '$999,999,990.00') AS "INGRESOS TOTALES",
    total_votos_actores AS "TOTAL VOTOS EMITIDOS"
FROM vw_reporte_financiero
WHERE anno IS NOT NULL -- Filtra registros si un club no ha tenido movimientos de pago todavía
ORDER BY anno DESC, nombre_club ASC;
